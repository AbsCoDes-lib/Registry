29  enum class RegistryHive {
30  ClassesRoot = static_cast<int>(0x80000000), // HKEY_CLASSES_ROOT
31  CurrentUser = static_cast<int>(0x80000001), // HKEY_CURRENT_USER
32  LocalMachine = static_cast<int>(0x80000002), // HKEY_LOCAL_MACHINE
33  Users = static_cast<int>(0x80000003), // HKEY_USERS
34  PerformanceData = static_cast<int>(0x80000004), // HKEY_PERFORMANCE_DATA
35  CurrentConfig = static_cast<int>(0x80000005), // HKEY_CURRENT_CONFIG
36  DynData = static_cast<int>(0x80000006), // HKEY_DYN_DATA
37  };

42  static HKEY Handle(RegistryHive hive) {
43  {
44  switch(hive) {
45  case RegistryHive::ClassesRoot: return HKEY_CLASSES_ROOT;
46  case RegistryHive::CurrentUser: return HKEY_CURRENT_USER;
47  case RegistryHive::LocalMachine: return HKEY_LOCAL_MACHINE;
48  case RegistryHive::Users: return HKEY_USERS;
49  case RegistryHive::PerformanceData: return HKEY_PERFORMANCE_DATA;
50  case RegistryHive::CurrentConfig: return HKEY_CURRENT_CONFIG;
51  case RegistryHive::DynData: return HKEY_DYN_DATA;
52  default: throw std::invalid_argument("Unsupported Windows registry hive.");
53  }
54  }
55  }
